1.Problem:
You are given a string (STR) of length N, consisting of only the lower case English alphabet.
Your task is to remove all the duplicate occurrences of characters in the string. Input: abcadeecfb

Ans

                function aLp(){
                    let STR=['a','b','c','a','d','e','e','c','f','b'];
                    let s=new Set();
                    for(let i=0;i< STR.length;i++){
                        s.add(STR[i]);
                    }
                    console.log(s);
                }
                aLp();

Output:
"abcdef"

abcadeecfb

            function mAp()
            {
                let s="abcadeecfb";
                let map=new Map();
                    for (let i=0; i< s.length;i++){
                        if(map.has(s[i])){
                            let value=map.get(s[i]);
                            map.set(s[i],value+1);
                        }
                        else{
                            map.set(s[i],1);
                         }
               
                    }
                    for(let [k,v] of map){
                    console.log(k +"-->"+v)
            }
        
        }
        mAp();
